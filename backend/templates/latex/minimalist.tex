% Custom commands
\newcommand{\name}[1]{
  \begin{center}
    {\huge\bfseries #1}
  \end{center}
  \vspace{6pt}
}

\newcommand{\contact}[1]{
  \begin{center}
    {\small\color{lightgray} #1}
  \end{center}
  \vspace{10pt}
}

\newcommand{\experience}[5]{
  \noindent
  \textbf{#1} \hfill {\small\color{lightgray} #2}\\
  {#3} \hfill {\small\color{lightgray} #4}
  \vspace{3pt}
  
  #5
  \vspace{8pt}
}

\newcommand{\education}[5]{
  \noindent
  \textbf{#1} \hfill {\small\color{lightgray} #2}\\
  {#3} \hfill {\small\color{lightgray} #4}
  
  {\small #5}
  \vspace{8pt}
}

\newcommand{\project}[4]{
  \noindent
  \textbf{#1} #2
  
  {\small\color{lightgray} #3}
  \vspace{3pt}
  
  #4
  \vspace{8pt}
}
