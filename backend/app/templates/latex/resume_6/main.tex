\documentclass{resume} % Use the custom resume.cls style

\usepackage[left=0.4 in,top=0.4in,right=0.4 in,bottom=0.4in]{geometry} % Document margins
\newcommand{\tab}[1]{\hspace{.2667\textwidth}\rlap{#1}} 
\newcommand{\itab}[1]{\hspace{0em}\rlap{#1}}

\address{\VAR{location|escape_tex}} % Your address
%\address{123 Pleasant Lane \\ City, State 12345} % Your secondary addess (optional)
\address{\VAR{phone|escape_tex} \\ \VAR{email|escape_tex}}  % Your phone number and email

\begin{document}

%----------------------------------------------------------------------------------------
%	OBJECTIVE
%----------------------------------------------------------------------------------------

\BLOCK{ if summary }
\begin{rSection}{OBJECTIVE}
{\VAR{summary|escape_tex}}
\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	EDUCATION SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if education }
\begin{rSection}{Education}

\BLOCK{ for edu in education }
{\bf \VAR{edu.degree|escape_tex} in \VAR{edu.field|escape_tex}} \hfill {\VAR{edu.start_date|escape_tex}\BLOCK{ if edu.start_date and edu.end_date } - \BLOCK{ endif }\VAR{edu.end_date|escape_tex}}
\\ 
\VAR{edu.institution|escape_tex}\BLOCK{ if edu.location }, \VAR{edu.location|escape_tex}\BLOCK{ endif }
\BLOCK{ if edu.gpa }
\\ CGPA: \VAR{edu.gpa|escape_tex}
\BLOCK{ endif }
\smallskip \\
\BLOCK{ endfor }

\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	TECHNICAL STRENGTHS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if skills }
\begin{rSection}{SKILLS}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{2ex}} p{11cm} }
\BLOCK{ for skill in skills }
\VAR{skill.category|escape_tex} & \VAR{skill.items|join(', ')|escape_tex} \\\\
\BLOCK{ endfor }
\end{tabular}

\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	WORK EXPERIENCE SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if experience }
\begin{rSection}{Work Experience}

\BLOCK{ for exp in experience }
\BLOCK{ if exp.description or (exp.highlights and exp.highlights|length > 0) }
\begin{rSubsection}{\VAR{exp.position|escape_tex}}{\VAR{exp.start_date|escape_tex}\BLOCK{ if exp.start_date and exp.end_date } - \BLOCK{ endif }\VAR{exp.end_date|escape_tex}}{\VAR{exp.company|escape_tex}}{\VAR{exp.location|escape_tex}}
\BLOCK{ if exp.description }
\item \VAR{exp.description|escape_tex}
\BLOCK{ endif }
\BLOCK{ for highlight in exp.highlights }
\item \VAR{highlight|escape_tex}
\BLOCK{ endfor }
\end{rSubsection}
\BLOCK{ else }
{\bf \VAR{exp.position|escape_tex}} \hfill {\VAR{exp.start_date|escape_tex}\BLOCK{ if exp.start_date and exp.end_date } - \BLOCK{ endif }\VAR{exp.end_date|escape_tex}}
\\
{\em \VAR{exp.company|escape_tex}} \hfill {\em \VAR{exp.location|escape_tex}}
\smallskip
\BLOCK{ endif }
\BLOCK{ endfor }

\end{rSection} 
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	PROJECTS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if projects }
\begin{rSection}{PROJECTS}

\BLOCK{ for proj in projects }
\begin{rSubsection}{\VAR{proj.name|escape_tex}}{}{}{}
\BLOCK{ if proj.description }
\item \VAR{proj.description|escape_tex}
\BLOCK{ endif }
\BLOCK{ if proj.technologies and proj.technologies|length > 0 }
\item Technologies: \VAR{proj.technologies|join(', ')|escape_tex}
\BLOCK{ endif }
\BLOCK{ if proj.link }
\item \VAR{proj.link|escape_tex}
\BLOCK{ endif }
\BLOCK{ if not proj.description and not (proj.technologies and proj.technologies|length > 0) and not proj.link }
\item 
\BLOCK{ endif }
\end{rSubsection} 
\BLOCK{ endfor }

\end{rSection} 
\BLOCK{ endif }
