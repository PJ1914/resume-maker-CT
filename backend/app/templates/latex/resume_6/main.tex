%----------------------------------------------------------------------------------------
%	PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%----------------------------------------------------------------------------------------

\documentclass{resume} % Use the custom resume.cls style

\usepackage[left=0.4 in,top=0.4in,right=0.4 in,bottom=0.4in]{geometry} % Document margins
\usepackage{hyperref} % For clickable links
\newcommand{\tab}[1]{\hspace{.2667\textwidth}\rlap{#1}} 
\newcommand{\itab}[1]{\hspace{0em}\rlap{#1}}
\renewcommand{\labelitemi}{$\vcenter{\hbox{\tiny$\bullet$}}$} % Custom small bullet

% Custom command for items without bullet points (prevents empty list error)
\newcommand{\rGenericItem}[4]{
  {\bf #1} \hfill {#2}
  \ifthenelse{\equal{#3}{}}{}{
   \\
   {\em #3} \hfill {\em #4}
  }
  \par\smallskip
}
\name{\VAR{full_name|escape_tex}} % Your name
\address{\VAR{location|escape_tex}} % Your address
\address{\VAR{phone|escape_tex} \\ \VAR{email|escape_tex}}  % Your phone number and email
\address{%
\BLOCK{ if linkedin }\href{\VAR{linkedin}}{LinkedIn}\BLOCK{ endif }%
\BLOCK{ if github }\BLOCK{ if linkedin } $\cdot$ \BLOCK{ endif }\href{\VAR{github}}{GitHub}\BLOCK{ endif }%
\BLOCK{ if leetcode } $\cdot$ \href{\VAR{leetcode}}{LeetCode}\BLOCK{ endif }%
\BLOCK{ if codechef } $\cdot$ \href{\VAR{codechef}}{CodeChef}\BLOCK{ endif }%
\BLOCK{ if hackerrank } $\cdot$ \href{\VAR{hackerrank}}{HackerRank}\BLOCK{ endif }%
\BLOCK{ if portfolio } $\cdot$ \href{\VAR{portfolio}}{Portfolio}\BLOCK{ endif }%
}

\begin{document}

%----------------------------------------------------------------------------------------
%	OBJECTIVE
%----------------------------------------------------------------------------------------

\BLOCK{ if summary }
\begin{rSection}{OBJECTIVE}
{\VAR{summary|escape_tex}}
\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	EDUCATION SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if education }
\begin{rSection}{Education}

\BLOCK{ for edu in education }
{\bf \VAR{edu.degree|escape_tex}\BLOCK{ if edu.field } in \VAR{edu.field|escape_tex}\BLOCK{ endif }} \hfill {\VAR{edu.start_date|escape_tex}\BLOCK{ if edu.start_date and edu.end_date } - \BLOCK{ endif }\VAR{edu.end_date|escape_tex}}
\\ 
\VAR{edu.school|escape_tex}\BLOCK{ if edu.location }, \VAR{edu.location|escape_tex}\BLOCK{ endif }
\BLOCK{ if edu.gpa }
\\ \BLOCK{ if edu.grade_type }\VAR{edu.grade_type|escape_tex}\BLOCK{ else }GPA/CGPA\BLOCK{ endif }: \VAR{edu.gpa|escape_tex}\BLOCK{ if edu.grade_type == 'Percentage' }\%\BLOCK{ endif }
\BLOCK{ endif }
\BLOCK{ if not loop.last }\smallskip \\\BLOCK{ endif }
\BLOCK{ endfor }

\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	TECHNICAL STRENGTHS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if skills }
\begin{rSection}{SKILLS}

\begin{tabular}{ @{} >{\bfseries}l @{\hspace{2ex}} p{11cm} }
\BLOCK{ for skill in skills }
\VAR{skill.category|escape_tex} & \VAR{skill.items|join(', ')|escape_tex} \\
\BLOCK{ endfor }
\end{tabular}

\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	WORK EXPERIENCE SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if experience }
\begin{rSection}{Work Experience}

\BLOCK{ for exp in experience }
\BLOCK{ if exp.description or (exp.highlights and exp.highlights|length > 0) }
\begin{rSubsection}{\VAR{exp.position|escape_tex}}{\VAR{exp.start_date|escape_tex}\BLOCK{ if exp.start_date and exp.end_date } - \BLOCK{ endif }\VAR{exp.end_date|escape_tex}}{\VAR{exp.company|escape_tex}}{\VAR{exp.location|escape_tex}}
\BLOCK{ if exp.description }
\item \VAR{exp.description|escape_tex}
\BLOCK{ endif }
\BLOCK{ for highlight in exp.highlights }
\item \VAR{highlight|escape_tex}
\BLOCK{ endfor }
\end{rSubsection}
\BLOCK{ else }
{\bf \VAR{exp.position|escape_tex}} \hfill {\VAR{exp.start_date|escape_tex}\BLOCK{ if exp.start_date and exp.end_date } - \BLOCK{ endif }\VAR{exp.end_date|escape_tex}}
\\
{\em \VAR{exp.company|escape_tex}} \hfill {\em \VAR{exp.location|escape_tex}}
\smallskip
\BLOCK{ endif }
\BLOCK{ endfor }

\end{rSection} 
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	PROJECTS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if projects }
\begin{rSection}{PROJECTS}

\BLOCK{ for proj in projects }
\begin{rSubsection}{\VAR{proj.name|escape_tex}}{}{}{}
\BLOCK{ if proj.description }
\item \VAR{proj.description|escape_tex}
\BLOCK{ endif }
\BLOCK{ if proj.technologies and proj.technologies|length > 0 }
\item Technologies: \VAR{proj.technologies|join(', ')|escape_tex}
\BLOCK{ endif }
\BLOCK{ if proj.link }
\item \VAR{proj.link|escape_tex}
\BLOCK{ endif }
\BLOCK{ if not proj.description and not (proj.technologies and proj.technologies|length > 0) and not proj.link }
\item 
\BLOCK{ endif }
\end{rSubsection} 
\BLOCK{ endfor }

\end{rSection} 
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	CERTIFICATIONS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if certifications and certifications|length > 0 }
\begin{rSection}{Certifications}
    \BLOCK{ for cert in certifications }
    \rGenericItem{\VAR{cert.name|escape_tex}}{\VAR{cert.date|escape_tex}}{\VAR{cert.issuer|escape_tex}}{}
    \BLOCK{ endfor }
\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	LANGUAGES SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if languages and languages|length > 0 }
\begin{rSection}{Languages}
    \BLOCK{ for lang in languages }
    \rGenericItem{\VAR{lang.language|escape_tex}}{\VAR{lang.proficiency|escape_tex}}{}{}
    \BLOCK{ endfor }
\end{rSection}
\BLOCK{ endif }

%----------------------------------------------------------------------------------------
%	ACHIEVEMENTS SECTION
%----------------------------------------------------------------------------------------

\BLOCK{ if achievements and achievements|length > 0 }
\begin{rSection}{Achievements} 
    \BLOCK{ for ach in achievements }
    \BLOCK{ if ach.description }
    \begin{rSubsection}{\VAR{ach.title|escape_tex}}{\VAR{ach.date|escape_tex}}{}{}
    \item \VAR{ach.description|escape_tex}
    \end{rSubsection}
    \BLOCK{ else }
    \rGenericItem{\VAR{ach.title|escape_tex}}{\VAR{ach.date|escape_tex}}{}{}
    \BLOCK{ endif }
    \BLOCK{ endfor }
\end{rSection}
\BLOCK{ endif }

\end{document}
