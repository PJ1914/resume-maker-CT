\documentclass[margin]{res}

%-------------------------------------------------------------------------------
%	PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%-------------------------------------------------------------------------------

% Default font is the helvetica postscript font
\usepackage{helvet}
\usepackage[hidelinks]{hyperref}
\usepackage{url}

% Increase text height
\textheight=700pt

\begin{document}

%-------------------------------------------------------------------------------
%	NAME AND ADDRESS SECTION
%-------------------------------------------------------------------------------
\name{\VAR{full_name|escape_tex}}

% Note that addresses can be used for other contact information:
% -phone numbers
% -email addresses
% -linked-in profile

\address{}%intentionally left blank 
\address{%
\BLOCK{ if email }\VAR{email|escape_tex}\BLOCK{ endif }%
\BLOCK{ if phone }\\\VAR{phone|escape_tex}\BLOCK{ endif }%
\BLOCK{ if location }\\\VAR{location|escape_tex}\BLOCK{ endif }%
\BLOCK{ if linkedin }\\\href{\VAR{linkedin}}{LinkedIn}\BLOCK{ endif }%
\BLOCK{ if github }\\\href{\VAR{github}}{GitHub}\BLOCK{ endif }%
\BLOCK{ if leetcode }\\\href{\VAR{leetcode}}{LeetCode}\BLOCK{ endif }%
\BLOCK{ if codechef }\\\href{\VAR{codechef}}{CodeChef}\BLOCK{ endif }%
\BLOCK{ if hackerrank }\\\href{\VAR{hackerrank}}{HackerRank}\BLOCK{ endif }%
\BLOCK{ if portfolio }\\\href{\VAR{portfolio}}{Portfolio}\BLOCK{ endif }%
}

% Uncomment to add a third address
%\address{Address 3 line 1\\Address 3 line 2\\Address 3 line 3}
%-------------------------------------------------------------------------------

\begin{resume}

%-------------------------------------------------------------------------------
%	EDUCATION SECTION
%-------------------------------------------------------------------------------
\BLOCK{ if education }
\section{EDUCATION}
\BLOCK{ for edu in education }
\textbf{\VAR{edu.school|escape_tex}}\BLOCK{ if edu.location }, \VAR{edu.location|escape_tex}\BLOCK{ endif }\\
{\sl \VAR{edu.degree|escape_tex}\BLOCK{ if edu.field }, \VAR{edu.field|escape_tex}\BLOCK{ endif }}, \VAR{edu.start_date|escape_tex}\BLOCK{ if edu.start_date and edu.end_date } - \BLOCK{ endif }\VAR{edu.end_date|escape_tex}\hfill \BLOCK{ if edu.gpa }\BLOCK{ if edu.grade_type }\VAR{edu.grade_type|escape_tex}\BLOCK{ else }GPA/CGPA\BLOCK{ endif }: \VAR{edu.gpa|escape_tex}\BLOCK{ if edu.grade_type == 'Percentage' }\%\BLOCK{ endif }\BLOCK{ endif }
\smallskip\\
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	SCHOLASTIC ACHEIVEMENTS
%-------------------------------------------------------------------------------
\BLOCK{ if achievements }
\section{ACHIEVEMENTS}
\BLOCK{ for ach in achievements }
\VAR{ach.title|escape_tex} \hfill \VAR{ach.date|escape_tex}\\
\BLOCK{ if ach.description }
\VAR{ach.description|escape_tex}\\
\BLOCK{ endif }
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	COMPUTER SKILLS SECTION
%-------------------------------------------------------------------------------
\BLOCK{ if skills }
\section{SKILLS}
\BLOCK{ for skill in skills }
\textbf{\VAR{skill.category|escape_tex}}: \VAR{skill.items|join(', ')|escape_tex}
\\
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	EXPERIENCE SECTION
%-------------------------------------------------------------------------------
% Modify the format of each position
\begin{format}
\title{l}\employer{r}\\
\dates{l}\location{r}\\
\body\\
\end{format}
%-------------------------------------------------------------------------------

\BLOCK{ if experience }
\section{EXPERIENCE}
\BLOCK{ for exp in experience }
\employer{\VAR{exp.company|escape_tex}}
\location{\VAR{exp.location|escape_tex}}
\dates{\VAR{exp.start_date|escape_tex}\BLOCK{ if exp.start_date and exp.end_date } -\BLOCK{ endif }\VAR{exp.end_date|escape_tex}}
\title{\textbf{\VAR{exp.position|escape_tex}}}
\begin{position}
\BLOCK{ if exp.description }
\VAR{exp.description|escape_tex}
\BLOCK{ endif }
\BLOCK{ if exp.highlights and exp.highlights|length > 0 }
\begin{itemize}
\BLOCK{ for highlight in exp.highlights }
\item \VAR{highlight|escape_tex}
\BLOCK{ endfor }
\end{itemize}
\BLOCK{ endif }
\end{position}
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	PROJECTS SECTION
%-------------------------------------------------------------------------------
\BLOCK{ if projects }
\section{PROJECTS}
\BLOCK{ for proj in projects }
\par
\textbf{\VAR{proj.name|escape_tex}}:
\BLOCK{ if proj.technologies }
Technologies: \VAR{proj.technologies|join(', ')|escape_tex}
\BLOCK{ endif }
\BLOCK{ if proj.description }
\begin{itemize}
\item \VAR{proj.description|escape_tex}
\end{itemize}
\BLOCK{ endif }
\BLOCK{ if proj.highlights and proj.highlights|length > 0 }
\begin{itemize}
\BLOCK{ for highlight in proj.highlights }
\item \VAR{highlight|escape_tex}
\BLOCK{ endfor }
\end{itemize}
\BLOCK{ endif }
\BLOCK{ if proj.link }
\VAR{proj.link|escape_tex}\\
\BLOCK{ endif }
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	CERTIFICATIONS
%-------------------------------------------------------------------------------
\BLOCK{ if certifications }
\section{CERTIFICATIONS}
\BLOCK{ for cert in certifications }
\textbf{\VAR{cert.name|escape_tex}} - \VAR{cert.issuer|escape_tex} \hfill \VAR{cert.date|escape_tex}\\
\BLOCK{ endfor }
\BLOCK{ endif }

%-------------------------------------------------------------------------------
%	LANGUAGES
%-------------------------------------------------------------------------------
\BLOCK{ if languages }
\section{LANGUAGES}
\BLOCK{ for lang in languages }
\textbf{\VAR{lang.language|escape_tex}}: \VAR{lang.proficiency|escape_tex}\\
\BLOCK{ endfor }
\BLOCK{ endif }

\end{resume}
\end{document}
