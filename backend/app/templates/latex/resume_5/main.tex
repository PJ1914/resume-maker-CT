  \BLOCK{ endfor }

  \BLOCK{ endfor }
  \BLOCK{ endif }

  \cvsection{Certifications}
  \BLOCK{ for cert in certifications }
  \par\textbf{\small \VAR{cert.name|escape_tex}}

  \par{\footnotesize\textit{\VAR{cert.issuer|escape_tex}}}
  \BLOCK{ endif }
  \medskip
  \BLOCK{ endfor }
  \BLOCK{ endif }
  
  \BLOCK{ if languages }
  \cvsection{Languages}
  \BLOCK{ for lang in languages }
  \cvtag{\VAR{lang.language|escape_tex}} (\VAR{lang.proficiency|escape_tex})
  \BLOCK{ endfor }
  \BLOCK{ endif }
  
  \BLOCK{ if achievements }
  \cvsection{Achievements}
  \BLOCK{ for ach in achievements }
  \par\textbf{\small \VAR{ach.title|escape_tex}}
